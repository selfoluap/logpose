```python
def _task_branch_name(task):
    slug = re.sub(r"[^\w\s-]", "", task["title"].lower().strip())
    slug = re.sub(r"[-\s]+", "-", slug).strip("-")
    return f"task-{task['id']}-{slug}"[:40].rstrip("-")


def _git(project_path, *args):
    subprocess.check_call(["git", *args], cwd=project_path)


def _current_branch(project_path):
    return subprocess.check_output(
        ["git", "branch", "--show-current"],
        cwd=project_path,
        text=True,
    ).strip()


def _is_pr_mode(project, config):
    project_path = os.path.abspath(project["path"])
    if project["pr_workflow"]:
        return True
    for d in config.get("pr_workflow", {}).get("dirs", []):
        root = os.path.abspath(os.path.expanduser(d))
        try:
            if os.path.commonpath([project_path, root]) == root:
                return True
        except ValueError:
            pass
    return False
```

import os
import tempfile

from logpose.db import get_db, project_add, project_update
from logpose.cli import _is_pr_mode, _task_branch_name


def test_project_pr_workflow_column_migrates_and_updates():
    with tempfile.NamedTemporaryFile() as f:
        conn = get_db(f.name)
        project = project_add(conn, "demo", "/tmp/demo", pr_workflow=True)

        assert project["pr_workflow"] == 1

        updated = project_update(conn, project["id"], pr_workflow=0)
        assert updated["pr_workflow"] == 0

        conn.close()


def test_pr_mode_matches_project_flag_or_config_dir():
    with tempfile.NamedTemporaryFile() as f:
        conn = get_db(f.name)
        flagged = project_add(conn, "flagged", "/tmp/anything", pr_workflow=True)
        matched = project_add(conn, "matched", "/tmp/work/app")
        plain = project_add(conn, "plain", "/opt/app")

        config = {"pr_workflow": {"dirs": ["/tmp/work"], "auto_pr": True}}

        assert _is_pr_mode(flagged, config)
        assert _is_pr_mode(matched, config)
        assert not _is_pr_mode(plain, config)

        conn.close()


def test_task_branch_name_is_capped_and_slugged():
    task = {"id": 12, "title": "Fix Login!!! With A Very Long Title That Keeps Going"}
    branch = _task_branch_name(task)

    assert branch.startswith("task-12-fix-login-with-a-very-long")
    assert len(branch) <= 40
